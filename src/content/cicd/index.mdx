      build-and-deploy:

          environment: production
          tag: ${{ github.sha }}
          github-app-id: ${{ vars.CI_BOT_APP_ID }}
          images: |
            [
              {
                "name": "application",
                "context": ".",
                "dockerfile": "./Dockerfile",
                "platforms": ["linux/amd64", "linux/arm64"]
              }
            ]
          chart-values: |
            [
              { "path": ".image", "image": "application" },
